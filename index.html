// ============================================================ // SONRISA NORTE - GOOGLE APPS SCRIPT API // Script para conectar Google Sheets con la aplicación web // ============================================================ // Configuración de las hojas de Google Sheets const CONFIG = { SHEET_USUARIOS: 'Usuarios', // Nombre de la hoja donde se guardan los usuarios SHEET_CITAS: 'Citas', // Nombre de la hoja donde se guardan las citas SHEET_SERVICIOS: 'Servicios', // Nombre de la hoja donde se guardan los servicios SHEET_HORARIOS: 'Horarios' // Nombre de la hoja donde se guardan los horarios }; // Función que se ejecuta cuando llega una petición GET al Web App function doGet(e) { return handleRequest(e); // Redirige la petición al manejador principal } // Función que se ejecuta cuando llega una petición POST al Web App function doPost(e) { return handleRequest(e); // Redirige la petición al manejador principal } // Función central que procesa todas las peticiones function handleRequest(e) { try { const params = e.parameter; // Obtiene los parámetros enviados en la petición const action = params.action; // Obtiene la acción que se quiere ejecutar let result; // Variable donde se guardará la respuesta // Selecciona qué función ejecutar según la acción recibida switch(action) { // ==================== USUARIOS ==================== case 'registrarUsuario': result = registrarUsuario(params); // Llama a la función de registro break; case 'loginUsuario': result = loginUsuario(params); // Llama a la función de login break; case 'verificarEmail': result = verificarEmail(params.email); // Verifica si existe el email break; case 'obtenerUsuarios': result = obtenerUsuarios(); // Obtiene todos los usuarios break; // ==================== CITAS ==================== case 'agendarCita': result = agendarCita(params); // Agenda una nueva cita break; case 'obtenerCitas': result = obtenerCitas(params.email); // Obtiene citas de un usuario break; case 'obtenerTodasCitas': result = obtenerTodasCitas(); // Obtiene todas las citas (admin) break; case 'obtenerHorariosOcupados': result = obtenerHorariosOcupados(params.fecha); // Obtiene horarios ocupados break; case 'actualizarEstadoCita': result = actualizarEstadoCita(params.id, params.estado); // Actualiza estado break; case 'cancelarCita': result = cancelarCita(params.id); // Cancela la cita break; // ==================== SERVICIOS ==================== case 'obtenerServicios': result = obtenerServicios(); // Obtiene los servicios break; // ==================== HORARIOS ==================== case 'obtenerHorarios': result = obtenerHorarios(); // Obtiene horarios disponibles break; default: result = { success: false, message: 'Acción no válida' }; // Acción desconocida } // Devuelve la respuesta en formato JSON return ContentService .createTextOutput(JSON.stringify(result)) // Convierte el resultado a texto JSON .setMimeType(ContentService.MimeType.JSON); // Indica que la respuesta es JSON } catch(error) { // Si ocurre un error return ContentService .createTextOutput(JSON.stringify({ success: false, message: 'Error en el servidor: ' + error.toString() // Mensaje de error })) .setMimeType(ContentService.MimeType.JSON); } }
